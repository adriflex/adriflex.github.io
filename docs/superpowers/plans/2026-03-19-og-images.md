# OG Images Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Auto-generate og:image PNGs for every project and lab page at build time using satori + resvg.

**Architecture:** Astro static endpoints (`src/pages/og/[type]/[slug].png.ts`) call a shared template function (`src/lib/og-template.ts`) that builds satori markup from content collection data. Cover images are read from disk as base64. The Base layout receives an optional `ogImage` prop and renders full Open Graph + Twitter Card meta tags.

**Tech Stack:** Astro 5, satori, @resvg/resvg-js, TypeScript

**Spec:** `docs/superpowers/specs/2026-03-19-og-images-design.md`

```typescript
import satori from 'satori';
import { Resvg } from '@resvg/resvg-js';
import fs from 'node:fs';
import path from 'node:path';

// Load fonts once
const silkscreenRegular = fs.readFileSync(
  path.resolve('src/assets/fonts/Silkscreen-Regular.ttf')
);
const silkscreenBold = fs.readFileSync(
  path.resolve('src/assets/fonts/Silkscreen-Bold.ttf')
);
const spaceMono = fs.readFileSync(
  path.resolve('src/assets/fonts/SpaceMono-Regular.ttf')
);

const FONTS = [
  { name: 'Silkscreen', data: silkscreenRegular, weight: 400 as const, style: 'normal' as const },
  { name: 'Silkscreen', data: silkscreenBold, weight: 700 as const, style: 'normal' as const },
  { name: 'Space Mono', data: spaceMono, weight: 400 as const, style: 'normal' as const },
];

// Colors from the design system
const COLORS = {
  bg: '#FEF6E4',
  navy: '#001858',
  para: '#172C66',
  pink: '#F582AE',
  cyan: '#8BD3DD',
  peach: '#F3D2C1',
};

// Badge config per type/status
const BADGES: Record<string, { label: string; bg: string }> = {
  published: { label: 'Published', bg: COLORS.pink },
  wip: { label: 'WIP', bg: COLORS.cyan },
  archived: { label: 'Archived', bg: '#CCD0D8' },
  lab: { label: 'Lab', bg: COLORS.peach },
};

export interface OgData {
  title: string;
  description?: string;
  tags: string[];
  cover?: string;       // relative path like "/images/foo.png"
  status?: string;      // "published" | "wip" | "archived"
  type: 'project' | 'lab';
}

/**
 * Load a cover image from public/ as a base64 data URI.
 * Returns undefined if the file doesn't exist.
 */
function loadCoverBase64(coverPath: string): string | undefined {
  try {
    const filePath = path.resolve('public', coverPath.replace(/^\//, ''));
    const buffer = fs.readFileSync(filePath);
    const ext = path.extname(filePath).replace('.', '');
    const mime = ext === 'jpg' ? 'image/jpeg' : `image/${ext}`;
    return `data:${mime};base64,${buffer.toString('base64')}`;
  } catch {
    return undefined;
  }
}

/**
 * Generate an og:image PNG buffer for the given page data.
 */
export async function generateOgImage(data: OgData): Promise<Buffer> {
  const coverDataUri = data.cover ? loadCoverBase64(data.cover) : undefined;
  const badgeKey = data.type === 'lab' ? 'lab' : (data.status || 'published');
  const badge = BADGES[badgeKey] || BADGES.published;

  // Build satori markup (React-like JSX objects)
  const markup = {
    type: 'div',
    props: {
      style: {
        width: '1200px',
        height: '630px',
        display: 'flex',
        flexDirection: 'column',
        backgroundColor: COLORS.bg,
        fontFamily: 'Space Mono',
        position: 'relative',
      },
      children: [
        // Main content row
        {
          type: 'div',
          props: {
            style: {
              display: 'flex',
              flex: 1,
            },
            children: [
              // Left side: text
              {
                type: 'div',
                props: {
                  style: {
                    width: '55%',
                    display: 'flex',
                    flexDirection: 'column',
                    justifyContent: 'center',
                    padding: '48px 48px 48px 56px',
                  },
                  children: [
                    // "ADRIFLEX" label
                    {
                      type: 'div',
                      props: {
                        style: {
                          fontFamily: 'Silkscreen',
                          fontSize: '14px',
                          color: COLORS.pink,
                          textTransform: 'uppercase',
                          letterSpacing: '3px',
                          marginBottom: '16px',
                        },
                        children: 'ADRIFLEX',
                      },
                    },
                    // Badge
                    {
                      type: 'div',
                      props: {
                        style: {
                          display: 'flex',
                          marginBottom: '20px',
                        },
                        children: [
                          {
                            type: 'div',
                            props: {
                              style: {
                                fontFamily: 'Silkscreen',
                                fontSize: '14px',
                                padding: '6px 16px',
                                backgroundColor: badge.bg,
                                color: COLORS.navy,
                              },
                              children: badge.label,
                            },
                          },
                        ],
                      },
                    },
                    // Title
                    {
                      type: 'div',
                      props: {
                        style: {
                          fontFamily: 'Silkscreen',
                          fontWeight: 700,
                          fontSize: '42px',
                          color: COLORS.navy,
                          lineHeight: 1.2,
                          marginBottom: '16px',
                        },
                        children: data.title,
                      },
                    },
                    // Description (projects only)
                    ...(data.description
                      ? [
                          {
                            type: 'div',
                            props: {
                              style: {
                                fontSize: '16px',
                                color: COLORS.para,
                                opacity: 0.6,
                                lineHeight: 1.6,
                                marginBottom: '20px',
                              },
                              children:
                                data.description.length > 100
                                  ? data.description.slice(0, 100) + '…'
                                  : data.description,
                            },
                          },
                        ]
                      : []),
                    // Tags
                    ...(data.tags.length > 0
                      ? [
                          {
                            type: 'div',
                            props: {
                              style: {
                                display: 'flex',
                                flexWrap: 'wrap',
                                gap: '8px',
                              },
                              children: data.tags.slice(0, 4).map((tag) => ({
                                type: 'div',
                                props: {
                                  style: {
                                    fontSize: '11px',
                                    padding: '4px 12px',
                                    border: `1px solid rgba(0,24,88,0.15)`,
                                    color: COLORS.navy,
                                    opacity: 0.6,
                                  },
                                  children: tag,
                                },
                              })),
                            },
                          },
                        ]
                      : []),
                  ],
                },
              },
              // Right side: cover or gradient
              {
                type: 'div',
                props: {
                  style: {
                    width: '45%',
                    display: 'flex',
                    alignItems: 'center',
                    justifyContent: 'center',
                    ...(coverDataUri
                      ? {}
                      : {
                          backgroundImage: `linear-gradient(135deg, ${COLORS.cyan} 0%, ${COLORS.pink} 100%)`,
                        }),
                  },
                  children: coverDataUri
                    ? [
                        {
                          type: 'img',
                          props: {
                            src: coverDataUri,
                            style: {
                              width: '100%',
                              height: '100%',
                              objectFit: 'cover',
                            },
                          },
                        },
                      ]
                    : [],
                },
              },
            ],
          },
        },
        // Bottom pink bar
        {
          type: 'div',
          props: {
            style: {
              height: '4px',
              backgroundColor: COLORS.pink,
              width: '100%',
            },
            children: [],
          },
        },
      ],
    },
  };

  // Generate SVG
  const svg = await satori(markup as any, {
    width: 1200,
    height: 630,
    fonts: FONTS,
  });

  // Convert SVG to PNG
  const resvg = new Resvg(svg, {
    fitTo: { mode: 'width', value: 1200 },
  });
  const pngData = resvg.render();
  return pngData.asPng();
}
```

```typescript
import type { APIRoute, GetStaticPaths } from 'astro';
import { getCollection } from 'astro:content';
import { generateOgImage } from '../../../lib/og-template';

export const getStaticPaths: GetStaticPaths = async () => {
  const projects = await getCollection('projects');
  return projects.map((entry) => ({
    params: { slug: entry.id.replace(/\.md$/, '') },
    props: { entry },
  }));
};

export const GET: APIRoute = async ({ props }) => {
  const { entry } = props as any;
  const png = await generateOgImage({
    title: entry.data.title,
    description: entry.data.description,
    tags: entry.data.tags,
    cover: entry.data.cover,
    status: entry.data.status,
    type: 'project',
  });

  return new Response(png, {
    headers: { 'Content-Type': 'image/png' },
  });
};
```
